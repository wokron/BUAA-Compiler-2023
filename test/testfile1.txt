int global = 0;
int func_cond() {
    global = global + 1;
    return 1;
}

    if (a && b < 10 || c > 2 && d) {
        printf("check pass\n");
    }
}

int main() {
    printf("21371326\n");
    int a = 3, b = 5, c = 10, d = 1;

    printf("%d\n", global);
    if (0 && func_cond()) {
        ;
    }

    printf("%d\n", global);
    if (1 || func_cond()) {
        ;
    }

    printf("%d\n", global);
    if (1 && func_cond() && 0 && func_cond()) {
        ;
    }

    printf("%d\n", global);
    if (0 || !func_cond() || 1 || !func_cond()) {
        ;
    }
    printf("%d\n", global);

    if (0 && func_cond() || func_cond() && 1) {
        ;
    }
    printf("%d\n", global);

    if (0 || !func_cond() && func_cond() && 1) {
        ;
    }
    printf("%d\n", global);

    if (a == 3 && b >= 5 || c <= 11 && !d) {
        printf("true\n");
    } else {

    }

        printf("true\n");
