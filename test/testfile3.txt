// testfile3: 多维数组和数组初始化

int main() {
    printf("21371326\n"); // 1

    int a[10 / 5];
    a[0] = 1;
    a[1] = 2;
    printf("a[0]=%d, a[1]=%d\n", a[0], a[1]); // 2

    int b[2][3] = {
        {1, 3, 5},
        {6, 4, 2}
    };

    int total = 0;
    for (int i = 0; i < 2; i = i+1) {
        int sum = 0;
        for (int j = 0; j < 3; j = j+1) {
            sum = sum + b[i][j];
        }
        printf("b[i][:]=%d\n", sum); // 3-4
        total = total + sum;
    }
    printf("b[:][:]=%d\n", total); // 5

    int c[4] = {1, 2, 3, 0};
    for (int i = 0; i < 4; i = i+1) {
        printf("c[%d]=%d\n", i, c[i]); // 6-9
    }

    int d[1] = {0};
    printf("d[%d]=%d\n", 0, d[0]);
    return 0;
}
